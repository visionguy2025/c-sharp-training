##### **SETUP COMMANDS** #####
    dotnet new webapi -n EcommerceAPI --framework net10.0;
    cd EcommerceAPI;
    dotnet new apicontroller -n AuthCTRL -o ./Controllers/Auth;
    dotnet new apicontroller -n CategoriesCTRL -o ./Controllers;
    dotnet new apicontroller -n ProductsCTRL -o ./Controllers;
    dotnet new apicontroller -n BrandCTRL -o ./Controllers;
    dotnet new class -n Base -o ./Entities;
    dotnet new class -n Category -o ./Entities;
    dotnet new class -n Product -o ./Entities;
    dotnet new class -n Brand -o ./Entities;
    dotnet new class -n GetCategoryDto -o ./Entities/DTOs/Category;
    dotnet new class -n CreateCategoryDto -o ./Entities/DTOs/Category;
    dotnet new class -n UpdateCategoryDto -o ./Entities/DTOs/Category;
    dotnet new class -n GetProductDto -o ./Entities/DTOs/Product;
    dotnet new class -n CreateProductDto -o ./Entities/DTOs/Product;
    dotnet new class -n UpdateProductDto -o ./Entities/DTOs/Product;
    dotnet new class -n GetBrandDto -o ./Entities/DTOs/Brand;
    dotnet new class -n CreateBrandDto -o ./Entities/DTOs/Brand;
    dotnet new class -n UpdateBrandDto -o ./Entities/DTOs/Brand;
    dotnet new class -n AppUser -o ./Entities/Auth;
    dotnet new class -n TokenOptions -o ./Entities/Auth;
    dotnet new class -n LoginDto -o ./Entities/DTOs/Auth;
    dotnet new class -n RegisterDto -o ./Entities/DTOs/Auth;
    dotnet new class -n EcomDbContext -o ./DAL;
    dotnet new class -n CategoryConfig -o ./DAL/Configurations;
    dotnet new class -n ProductConfig -o ./DAL/Configurations;
    dotnet new class -n BrandConfig -o ./DAL/Configurations;
    dotnet new interface -n ICategoryRepository -o ./DAL/Repositories/Abstract;
    dotnet new interface -n IProductRepository -o ./DAL/Repositories/Abstract;
    dotnet new interface -n IBrandRepository -o ./DAL/Repositories/Abstract;
    dotnet new class -n CategoryRepository -o ./DAL/Repositories/Concrete/EntityFrameworkCore;
    dotnet new class -n ProductRepository -o ./DAL/Repositories/Concrete/EntityFrameworkCore;
    dotnet new class -n BrandRepository -o ./DAL/Repositories/Concrete/EntityFrameworkCore;
    dotnet new class -n GetCategoryValidation -o ./Validators/Category;
    dotnet new class -n CreateCategoryValidation -o ./Validators/Category;
    dotnet new class -n UpdateCategoryValidation -o ./Validators/Category;
    dotnet new class -n GetProductValidation -o ./Validators/Product;
    dotnet new class -n CreateProductValidation -o ./Validators/Product;
    dotnet new class -n UpdateProductValidation -o ./Validators/Product;
    dotnet new class -n GetBrandValidation -o ./Validators/Brand;
    dotnet new class -n CreateBrandValidation -o ./Validators/Brand;
    dotnet new class -n UpdateBrandValidation -o ./Validators/Brand;
    dotnet new class -n CategoryMapper -o ./Profiles;
    dotnet new class -n ProductMapper -o ./Profiles;
    dotnet new class -n BrandMapper -o ./Profiles;
    mkdir ./DAL/Repositories/Concrete/Dapper;
    New-Item ./DAL/Repositories/Concrete/Dapper/empty.txt;
    dotnet add package Microsoft.EntityFrameworkCore --version 10.0.1;
    dotnet add package Microsoft.EntityFrameworkCore.SqlServer --version 10.0.1;
    dotnet add package Microsoft.EntityFrameworkCore.Tools --version 10.0.1;
    dotnet add package Microsoft.EntityFrameworkCore.Design --version 10.0.1;
    dotnet add package Swashbuckle.AspNetCore --version 10.1.2;
    dotnet add package Microsoft.AspNetCore.Identity.EntityFrameworkCore --version 10.0.1;
    dotnet add package Microsoft.AspNetCore.Authentication.JwtBearer --version 10.0.1;
    dotnet add package FluentValidation --version 12.1.1;
    dotnet add package FluentValidation.AspNetCore --version 11.3.1;
    dotnet add package AutoMapper --version 12.0.1;
    dotnet add package AutoMapper.Extensions.Microsoft.DependencyInjection --version 12.0.1;
    dotnet new tool-manifest;
    dotnet tool install dotnet-ef --version 10.0.1;
    dotnet build;


##### **APP USAGE** #####
    dotnet run;
    dotnet ef migrations add Init -p ./ -o ./DAL/Migrations;
    dotnet ef database update;
